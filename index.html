CSS

.card {
width: 300px;
height: 200px;
perspective: 1000px;
margin: 15px;
object-fit: cover;
}

.card-inner {
width: 100%;
height: 100%;
transform-style: preserve-3d;
transition: transform 0.6s;
object-fit: cover;
}

.card-inner:hover {
transform: rotateY(180deg);
}

.card-front, .card-back { /* note the syntax where the comma is*/
width: 100%;
height: 100%;
position: absolute; /* This keeps the card back and front in one place when being hovered over so that it looks like an actual card*/
backface-visibility: hidden; /* Prevents flickering during the flip */
display: flex;
flex-direction: column;
justify-content: center;
align-items: center;
box-shadow: 0 4px 8px rgba(0,0,0,0.1);
border-radius: 10px;
background-color: #fff;
padding: 20px;
object-fit: cover;
}


.card-back {
transform: rotateY(180deg);
object-fit: cover;
}

.card-front img {
max-width: 100%; /* Ensure the image fits within its container */
max-height: 100%; /* Ensure the image fits within its container */
border-radius: 10px; /* Match the card's border-radius */
}

.card-back img {
max-width: 100%; /* Ensure the image fits within its container */
max-height: 80%; /* Ensure the image fits within its container */
border-radius: 10px; /* Match the card's border-radius */
margin-top: 19px;
}
